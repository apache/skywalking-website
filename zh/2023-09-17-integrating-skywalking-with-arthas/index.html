е°Ҷ Apache SkyWalking дёҺ Arthas йӣҶжҲҗ

жң¬зҜҮж–Үз« жј”зӨәеҰӮдҪ•е°Ҷ Arthas йӣҶжҲҗеҲ° SkyWalking дёӯпјҢе№¶еҖҹжӯӨеңәжҷҜи®Ёи®ә SkyWalking дёӯзҡ„еёёи§Ғжү©еұ•зӮ№гҖӮ
By йӯҸзҝ” | Sunday, September 17, 2023

иғҢжҷҜд»Ӣз»Қ

Arthas жҳҜдёҖж¬ҫеёёз”Ёзҡ„ Java иҜҠж–ӯе·Ҙе…·пјҢжҲ‘д»¬еҸҜд»ҘеңЁ SkyWalking зӣ‘жҺ§еҲ°жңҚеҠЎејӮеёёеҗҺпјҢйҖҡиҝҮ Arthas иҝӣдёҖжӯҘеҲҶжһҗе’ҢиҜҠж–ӯд»Ҙеҝ«йҖҹе®ҡдҪҚй—®йўҳгҖӮ

еңЁ Arthas е®һйҷ…дҪҝз”ЁдёӯпјҢйҖҡеёёз”ұејҖеҸ‘дәәе‘ҳжӢ·иҙқжҲ–иҖ…дёӢиҪҪе®үиЈ…еҢ…еҲ°жңҚеҠЎеҜ№еә”зҡ„VMжҲ–иҖ…е®№еҷЁдёӯпјҢattach еҲ°еҜ№еә”зҡ„ Java иҝӣзЁӢиҝӣиЎҢй—®йўҳжҺ’жҹҘгҖӮиҝҷдёҖиҝҮзЁӢдёҚеҸҜйҒҝе…Қзҡ„дјҡйҖ жҲҗжңҚеҠЎеҷЁж•Ҹж„ҹиҝҗз»ҙдҝЎжҒҜзҡ„жү©ж•ЈпјҢ иҖҢдё”еңЁеҲҶз§’еҝ…дәүзҡ„й—®йўҳжҺ’жҹҘиҝҮзЁӢдёӯпјҢиҝҷдәӣз№Ғзҗҗзҡ„ж“ҚдҪңж— з–‘дјҡжөӘиҙ№еӨ§йҮҸж—¶й—ҙгҖӮ

SkyWalking Java Agent дјҙйҡҸ Java жңҚеҠЎдёҖиө·еҗҜеҠЁпјҢе№¶е®ҡжңҹдёҠжҠҘжңҚеҠЎгҖҒе®һдҫӢдҝЎжҒҜз»ҷOAP ServerгҖӮжҲ‘д»¬еҸҜд»ҘеҖҹеҠ© SkyWalking Java Agent зҡ„жҸ’д»¶еҢ–иғҪеҠӣпјҢејҖеҸ‘дёҖдёӘ Arthas жҺ§еҲ¶жҸ’д»¶пјҢ з”ұиҜҘжҸ’д»¶з®ЎзҗҶ Arthas иҝҗиЎҢз”ҹе‘Ҫе‘ЁжңҹпјҢйҖҡиҝҮйЎөйқўеҢ–зҡ„ж–№ејҸпјҢе®ҢжҲҗArthasзҡ„еҗҜеҠЁдёҺеҒңжӯўгҖӮжңҖз»Ҳе®һзҺ°ж•ҲжһңеҸҜд»ҘеҸӮиҖғдёӢеӣҫпјҡ

иҰҒе®ҢжҲҗдёҠиҝ°еҠҹиғҪпјҢжҲ‘д»¬йңҖиҰҒе®һзҺ°д»ҘдёӢеҮ дёӘе…ій”®зӮ№пјҡ

  1. ејҖеҸ‘ agent arthas-control-pluginпјҢжү§иЎҢ arthas зҡ„еҗҜеҠЁдёҺеҒңжӯўе‘Ҫд»Ө
  2. ејҖеҸ‘ oap arthas-controller-module пјҢдёӢеҸ‘жҺ§еҲ¶е‘Ҫд»Өз»ҷ arthas agent plugin
  3. е®ҡеҲ¶ skywalking-ui, иҝһжҺҘ arthas-tunnel-serverпјҢеҸ‘йҖҒ arthas е‘Ҫд»Өе№¶иҺ·еҸ–жү§иЎҢз»“жһң

д»ҘдёҠеҗ„дёӘжЁЎеқ—д№Ӣй—ҙзҡ„дәӨдә’жөҒзЁӢеҰӮдёӢеӣҫжүҖзӨәпјҡ

жҺҘзқҖпјҢжҲ‘д»¬зңӢдёӢ CommandListener.java зҡ„е®һзҺ°пјҢCommandListener е®һзҺ°дәҶ BootService жҺҘеҸЈпјҢ е№¶йҖҡиҝҮ resources/META-INF/services дёӢзҡ„ж–Үд»¶жҡҙйңІз»ҷ ServiceLoaderгҖӮ

BootService зҡ„е®ҡд№үеҰӮдёӢпјҢе…ұжңүprepare()гҖҒboot()гҖҒonComplete()гҖҒshutdown()еҮ дёӘж–№жі•пјҢиҝҷеҮ дёӘж–№жі•еҲҶеҲ«еҜ№еә”жҸ’д»¶з”ҹе‘Ҫе‘Ёжңҹзҡ„дёҚеҗҢйҳ¶ж®өгҖӮ

public interface BootService {
    void prepare() throws Throwable;

    void boot() throws Throwable;

    void onComplete() throws Throwable;

    void shutdown() throws Throwable;

    default int priority() {
        return 0;
    }
}

getCommandж–№жі•дёӯе®ҡд№үдәҶstartгҖҒstopзҡ„еӨ„зҗҶйҖ»иҫ‘пјҢеҲҶеҲ«еҜ№еә”йЎөйқўдёҠзҡ„ connect е’Ң disconnect ж“ҚдҪңгҖӮ иҝҷдёӨдёӘ command жңүеҲҶеҲ«иҪ¬з»ҷ ArthasCtl зҡ„ startArthas е’Ң stopArthas дёӨдёӘж–№жі•еӨ„зҗҶпјҢз”ЁжқҘжҺ§еҲ¶ arthas зҡ„еҗҜеҒңгҖӮ

еңЁ startArthas ж–№жі•дёӯпјҢеҗҜеҠЁarthas-core.jar е№¶дҪҝз”Ё skywalking-agent зҡ„ serviceName е’Ң instanceName жіЁеҶҢиҝһжҺҘиҮій…ҚзҪ®ж–Үд»¶дёӯжҢҮе®ҡзҡ„arthas-tunnel-serverгҖӮ

ArthasCtl йҖ»иҫ‘еҸӮиҖғиҮӘ Arthas зҡ„ BootStrap.java пјҢз”ұдәҺдёҚжҳҜжң¬зҜҮж–Үз« зҡ„йҮҚзӮ№пјҢиҝҷйҮҢдёҚеҶҚиөҳиҝ°пјҢж„ҹе…ҙи¶Јзҡ„е°ҸдјҷдјҙеҸҜд»ҘиҮӘиЎҢжҹҘзңӢгҖӮ

switch (commandResponse.getCommand()) {
    case START:
        if (alreadyAttached()) {
            LOGGER.warn("arthas already attached, no need start again");
            return;
        }
        try {
            arthasTelnetPort = SocketUtils.findAvailableTcpPort();
            ArthasCtl.startArthas(PidUtils.currentLongPid(), arthasTelnetPort);
        } catch (Exception e) {
            LOGGER.info("error when start arthas", e);
        }
        break;
    case STOP:
        if (!alreadyAttached()) {
            LOGGER.warn("no arthas attached, no need to stop");
            return;
        }
        try {
            ArthasCtl.stopArthas(arthasTelnetPort);
            arthasTelnetPort = null;
        } catch (Exception e) {
            LOGGER.info("error when stop arthas", e);
        }
        break;
}

е…¶е®һиҝҷжҳҜskywalking-agentзҡ„дёҖдёӘе°Ҹйӯ”жі•пјҢз”ұдәҺ agent-core жңҖз»Ҳдјҡиў«жү“еҢ…иҝӣ skywalking-agent.jarпјҢеҗҜеҠЁж—¶з”ұзі»з»ҹзұ»иЈ…иҪҪеҷЁпјҲжҲ–иҖ…е…¶д»–зҲ¶зә§зұ»иЈ…иҪҪеҷЁпјүзӣҙжҺҘиЈ…иҪҪпјҢ дёәдәҶйҳІжӯўжүҖдҫқиө–зҡ„зұ»еә“е’Ңиў«зӣ‘жҺ§жңҚеҠЎзҡ„зұ»еҸ‘з”ҹзүҲжң¬еҶІзӘҒпјҢagent ж ёеҝғд»Јз ҒеңЁжү“еҢ…ж—¶дҪҝз”ЁдәҶmaven-shade-plugin, иҜҘжҸ’д»¶дјҡеңЁ maven package йҳ¶ж®өж”№еҸҳ grpc дҫқиө–зҡ„еҢ…еҗҚпјҢ жҲ‘д»¬еңЁжәҗд»Јз ҒйҮҢзңӢеҲ°зҡ„жҳҜ io.grpc.ChannelпјҢе…¶е®һеңЁзңҹжӯЈиҝҗиЎҢж—¶е·Із»Ҹиў«ж”№жҲҗдәҶ org.apache.skywalking.apm.dependencies.io.grpc.ChannelпјҢиҝҷдҫҝеҸҜи§ЈйҮҠдёҠйқўзј–иҜ‘жҠҘй”ҷзҡ„еҺҹеӣ гҖӮ

public class CommandQueue {

    private static final Map<String, Command> COMMANDS = new ConcurrentHashMap<>();

    // produce by connectгҖҒdisconnect
    public static void produceCommand(String serviceName, String instanceName, Command command) {
        COMMANDS.put(serviceName + instanceName, command);
    }

    // consume by agent getCommand task
    public static Optional<Command> consumeCommand(String serviceName, String instanceName) {
        return Optional.ofNullable(COMMANDS.remove(serviceName + instanceName));
    }
}
